## List Customers

Retrieve all customers for a specific branch.

<CodeGroup>
```http Request
POST /clientes/getClientes
Content-Type: application/json

{
  "id_sucursal": 1
}
```

```json Response
{
  "data": [
    {
      "id": 1,
      "nombre": "Juan Pérez",
      "numero_documento": "12345678",
      "celular": "987654321",
      "tipo_cliente": "Natural",
      "tipo_pago": "Contado",
      "incluye_igv": 1,
      "id_distrito": 150101,
      "distrito_nombre": "Lima",
      "plan_nombre": "Básico",
      "lat": "-12.0464",
      "lon": "-77.0428",
      "extra_domdom": 5,
      "extra_domage": 3,
      "extra_agedom": 3
    }
  ]
}
```
</CodeGroup>

### Request Parameters

<ParamField body="id_sucursal" type="integer" required>
  Branch ID to filter customers
</ParamField>

### Response Fields

<ResponseField name="data" type="array">
  Array of customer objects

  <Expandable title="Customer Object">
    <ResponseField name="id" type="integer">
      Customer unique identifier
    </ResponseField>

    <ResponseField name="nombre" type="string">
      Customer full name
    </ResponseField>

    <ResponseField name="numero_documento" type="string">
      Document number (DNI/RUC)
    </ResponseField>

    <ResponseField name="celular" type="string">
      Mobile phone number
    </ResponseField>

    <ResponseField name="tipo_cliente" type="string">
      Customer type (e.g., "Natural", "Jurídico")
    </ResponseField>

    <ResponseField name="tipo_pago" type="string">
      Payment type
    </ResponseField>

    <ResponseField name="incluye_igv" type="integer">
      Includes IGV (tax): 1 = yes, 0 = no
    </ResponseField>

    <ResponseField name="id_distrito" type="integer">
      District ID
    </ResponseField>

    <ResponseField name="distrito_nombre" type="string">
      District name
    </ResponseField>

    <ResponseField name="plan_nombre" type="string">
      Price plan name
    </ResponseField>

    <ResponseField name="lat" type="string">
      Latitude coordinate
    </ResponseField>

    <ResponseField name="lon" type="string">
      Longitude coordinate
    </ResponseField>

    <ResponseField name="extra_domdom" type="integer">
      Extra charge for home-to-home delivery
    </ResponseField>

    <ResponseField name="extra_domage" type="integer">
      Extra charge for home-to-agency delivery
    </ResponseField>

    <ResponseField name="extra_agedom" type="integer">
      Extra charge for agency-to-home delivery
    </ResponseField>
  </Expandable>
</ResponseField>

## Search Customer

Search for customers by document number or name.

<CodeGroup>
```http Request
GET /clientes/buscarCliente?busqueda=12345678
```

```json Response
{
  "data": [
    {
      "id": 1,
      "nombre": "Juan Pérez",
      "numero_documento": "12345678"
    },
    {
      "id": 2,
      "nombre": "Juan García",
      "numero_documento": "87654321"
    }
  ]
}
```
</CodeGroup>

### Query Parameters

<ParamField query="busqueda" type="string" required>
  Search term (document number or name)
</ParamField>

### Response Fields

<ResponseField name="data" type="array">
  Array of matching customers

  <Expandable title="Customer Object">
    <ResponseField name="id" type="integer">
      Customer unique identifier
    </ResponseField>

    <ResponseField name="nombre" type="string">
      Customer full name
    </ResponseField>

    <ResponseField name="numero_documento" type="string">
      Document number (DNI/RUC)
    </ResponseField>
  </Expandable>
</ResponseField>
